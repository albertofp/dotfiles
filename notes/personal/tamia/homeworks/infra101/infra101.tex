\begin{frame}{Cloud}
\begin{block}{What?}
Someone else's infrastructure (usually paid)
\end{block}
\begin{block}{How?}
Use a UI, CLI, or code to provision/manage resources and tear them down when needed.
\end{block}
\begin{block}{Why?}
\begin{itemize}
\item Pay-as-you-go
\item Reliability
\item Scaling
\item Security
\end{itemize}
\end{block}
\begin{block}{Examples}
AWS, Azure, Google Cloud, Hetzner
\end{block}
\end{frame}

\begin{frame}{Infrastructure as Code}
\textbf{Scenario 1:} Expensive, complex deployment; ensure complete teardown and recovery.
\textbf{Scenario 2:} Team work with repetitive resources; minimize manual errors.
\medskip
\textbf{Solution:}
\begin{itemize}
\item Declarative language for desired state
\item State logging
\item Version control
\item Self documenting and modularized
\end{itemize}

\begin{frame}{Containerization}
\textbf{Problem:} "But it works on my machine!"\\[1ex]
\textbf{Solution:} Ship your whole machine.
\medskip
Docker:
\begin{itemize}
\item Builds a container from your code
\item Runs on any Docker-enabled machine
\item Easy to deploy and remove
\end{itemize}
